# Create a varible with name 'num2' which has input the number and convert this inputing text to type 'INT' use a func 'INPUT_INT'
VAR num2 = INPUT_INT()

# Print request of take the token of operate use a func 'PRINT'
PRINT("Please enter the token of operation: ")

# Now, create a varible with name 'token' which has a inputing text with token for our operation use a func 'INPUT'
VAR token = INPUT()

# Now, create a conditional statement loop and do proces use a funcs 'IF, ELIF, ELSE, END, PRINT' and token which we take earlier
IF token == 1 THEN
    PRINT(num1 + num2)
ELIF token == "-" THEN
    PRINT(num1 - num2)
ELIF token == "/" THEN
    PRINT(num1 / num2)
ELIF token == "*" THEN
    PRINT(num1 * num2)
ELIF token == "**" THEN
    PRINT(num1**num2)
ELSE
    PRINT("Error") # Print error with user give not correct data
END 
# Close the conditional statement loop
